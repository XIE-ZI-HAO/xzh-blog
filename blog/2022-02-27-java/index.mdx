## java 练习

```js
import java.util.Scanner; //导入的类
public class LX {
    public static  void main(String[] args)
    {
        Scanner sc=new Scanner(System.in);
        System.out.println("请输入第一个整数：");
        int num1 = sc.nextInt();
        System.out.println("请输入第二个整数：");
        int num2 = sc.nextInt();
        int max =num1>num2?num1:num2; //如果num1大于num2，num1将和num2作比较
        System.out.println("请输入整数最大的是："+max);
    }
}
```

```js
import java.util.Scanner;
public class W {
    public static void main(String[] args){
        Scanner input = new Scanner(System.in);
        String a=input.nextLine();  //nextline表示将空格后的也打印
        String b=input.next();  //next表示不打印空格后的
        System.out.println(a);
        System.out.println(b);
    }
}

```
